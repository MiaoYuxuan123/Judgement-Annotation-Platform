**2.建表SQL**

\-- ==================== 用户权限模块 ====================

\-- 用户表

CREATE TABLE \`sys_user\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '用户ID（主键）',

&nbsp;   \`username\` VARCHAR(50) NOT NULL COMMENT '登录账号（唯一）',

&nbsp;   \`password_hash\` VARCHAR(255) NOT NULL COMMENT '密码',

&nbsp;   \`real_name\` VARCHAR(100) NOT NULL COMMENT '真实姓名',

&nbsp;   \`phone\` VARCHAR(20) DEFAULT NULL COMMENT '手机号',

&nbsp;   \`status\` TINYINT NOT NULL DEFAULT 1 COMMENT '状态：1正常 0禁用',

&nbsp;   \`role_id\` INT UNSIGNED NOT NULL COMMENT '全局角色ID（外键→sys_role.id）',

&nbsp;   \`created_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

&nbsp;   \`updated_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   UNIQUE KEY \`uk_username\` (\`username\`),

&nbsp;   KEY \`idx_status\` (\`status\`),

&nbsp;   KEY \`idx_role\` (\`role_id\`),

&nbsp;   CONSTRAINT \`fk_user_role\` FOREIGN KEY (\`role_id\`) REFERENCES \`sys_role\` (\`id\`) ON DELETE RESTRICT

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='用户表';

\-- 全局角色表

CREATE TABLE \`sys_role\` (

&nbsp;   \`id\` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '角色ID（主键）',

&nbsp;   \`role_code\` VARCHAR(30) NOT NULL COMMENT '角色代码：SUPER_ADMIN/TASK_CREATOR/USER',

&nbsp;   \`role_name\` VARCHAR(50) NOT NULL COMMENT '角色显示名称',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   UNIQUE KEY \`uk_role_code\` (\`role_code\`)

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='全局角色表';

\-- ==================== 配置中心模块 ====================

\-- 指南版本表

CREATE TABLE \`guide_version\` (

&nbsp;   \`id\` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '指南版本ID（主键）',

&nbsp;   \`version_name\` VARCHAR(100) NOT NULL COMMENT '版本名称',

&nbsp;   \`description\` TEXT COMMENT '版本说明',

&nbsp;   \`is_active\` TINYINT NOT NULL DEFAULT 0 COMMENT '是否为当前启用版本：1是 0否',

&nbsp;   \`created_by\` BIGINT UNSIGNED NOT NULL COMMENT '创建人ID（外键→sys_user.id）',

&nbsp;   \`created_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_active\` (\`is_active\`),

&nbsp;   CONSTRAINT \`fk_guide_version_user\` FOREIGN KEY (\`created_by\`) REFERENCES \`sys_user\` (\`id\`) ON DELETE RESTRICT

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='指南版本表';

\-- 一级标签表

CREATE TABLE \`label_l1\` (

&nbsp;   \`id\` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '一级标签ID（主键）',

&nbsp;   \`guide_version_id\` INT UNSIGNED NOT NULL COMMENT '所属指南版本ID（外键→guide_version.id）',

&nbsp;   \`name\` VARCHAR(50) NOT NULL COMMENT '标签全称',

&nbsp;   \`abbr\` VARCHAR(10) NOT NULL COMMENT '标签简称',

&nbsp;   \`sort_order\` INT NOT NULL DEFAULT 0 COMMENT '显示顺序',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   UNIQUE KEY \`uk_version_abbr\` (\`guide_version_id\`, \`abbr\`),

&nbsp;   CONSTRAINT \`fk_label_l1_version\` FOREIGN KEY (\`guide_version_id\`) REFERENCES \`guide_version\` (\`id\`) ON DELETE CASCADE

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='一级标签表';

\-- 二级标签表

CREATE TABLE \`label_l2\` (

&nbsp;   \`id\` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '二级标签ID（主键）',

&nbsp;   \`guide_version_id\` INT UNSIGNED NOT NULL COMMENT '所属指南版本ID（外键→guide_version.id）',

&nbsp;   \`parent_l1_id\` INT UNSIGNED NOT NULL COMMENT '所属GM一级标签ID（外键→label_l1.id）',

&nbsp;   \`name\` VARCHAR(50) NOT NULL COMMENT '二级标签全称',

&nbsp;   \`abbr\` VARCHAR(20) NOT NULL COMMENT '二级标签简称',

&nbsp;   \`sort_order\` INT NOT NULL DEFAULT 0 COMMENT '显示顺序',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_parent\` (\`parent_l1_id\`),

&nbsp;   CONSTRAINT \`fk_label_l2_version\` FOREIGN KEY (\`guide_version_id\`) REFERENCES \`guide_version\` (\`id\`) ON DELETE CASCADE,

&nbsp;   CONSTRAINT \`fk_label_l2_parent\` FOREIGN KEY (\`parent_l1_id\`) REFERENCES \`label_l1\` (\`id\`) ON DELETE CASCADE

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='二级标签表（仅GM）';

\-- 关系类型表

CREATE TABLE \`relation_type\` (

&nbsp;   \`id\` INT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '关系类型ID（主键）',

&nbsp;   \`guide_version_id\` INT UNSIGNED NOT NULL COMMENT '所属指南版本ID（外键→guide_version.id）',

&nbsp;   \`name\` VARCHAR(50) NOT NULL COMMENT '关系名称',

&nbsp;   \`abbr\` CHAR(1) NOT NULL COMMENT '关系简称',

&nbsp;   \`is_binary\` TINYINT NOT NULL DEFAULT 1 COMMENT '是否为二元关系：1是 0否',

&nbsp;   \`sort_order\` INT NOT NULL DEFAULT 0 COMMENT '显示顺序',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   UNIQUE KEY \`uk_version_abbr\` (\`guide_version_id\`, \`abbr\`),

&nbsp;   CONSTRAINT \`fk_relation_type_version\` FOREIGN KEY (\`guide_version_id\`) REFERENCES \`guide_version\` (\`id\`) ON DELETE CASCADE

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='关系类型表';

\-- ==================== 任务管理模块 ====================

\-- 任务表

CREATE TABLE \`task\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '任务ID（主键）',

&nbsp;   \`title\` VARCHAR(200) NOT NULL COMMENT '任务名称',

&nbsp;   \`description\` TEXT COMMENT '任务描述',

&nbsp;   \`status\` VARCHAR(20) NOT NULL DEFAULT '待标注' COMMENT '任务状态：待标注/待裁定/可导出',

&nbsp;   \`guide_version_id\` INT UNSIGNED NOT NULL COMMENT '使用的指南版本ID（外键→guide_version.id）',

&nbsp;   \`label_config_snapshot\` JSON NOT NULL COMMENT '标签配置快照',

&nbsp;   \`creator_id\` BIGINT UNSIGNED NOT NULL COMMENT '创建者ID（外键→sys_user.id）',

&nbsp;   \`created_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

&nbsp;   \`stage_changed_at\` DATETIME DEFAULT NULL COMMENT '阶段最近变更时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_creator\` (\`creator_id\`),

&nbsp;   KEY \`idx_status\` (\`status\`),

&nbsp;   KEY \`idx_guide_version\` (\`guide_version_id\`),

&nbsp;   CONSTRAINT \`fk_task_creator\` FOREIGN KEY (\`creator_id\`) REFERENCES \`sys_user\` (\`id\`) ON DELETE RESTRICT,

&nbsp;   CONSTRAINT \`fk_task_guide_version\` FOREIGN KEY (\`guide_version_id\`) REFERENCES \`guide_version\` (\`id\`) ON DELETE RESTRICT

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='任务表';

\-- 全局文件表

CREATE TABLE \`global_document\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '全局文件ID（主键）',

&nbsp;   \`file_name\` VARCHAR(255) NOT NULL COMMENT '文件名',

&nbsp;   \`file_path\` VARCHAR(500) NOT NULL COMMENT '文件存储路径',

&nbsp;   \`file_type\` VARCHAR(10) NOT NULL COMMENT '文件类型：pdf/docx/txt',

&nbsp;   \`extracted_text\` LONGTEXT NOT NULL COMMENT '提取的纯文本内容',

&nbsp;   \`uploaded_by_id\` BIGINT UNSIGNED NOT NULL COMMENT '上传者ID（外键→sys_user.id，仅限SUPER_ADMIN）',

&nbsp;   \`uploaded_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '上传时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_uploader\` (\`uploaded_by_id\`),

&nbsp;   KEY \`idx_file_name\` (\`file_name\`(100)),

&nbsp;   CONSTRAINT \`fk_global_doc_user\` FOREIGN KEY (\`uploaded_by_id\`) REFERENCES \`sys_user\` (\`id\`) ON DELETE RESTRICT

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='全局文件库';

\-- 任务文档表

CREATE TABLE \`task_document\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '任务文档关联ID（主键）',

&nbsp;   \`task_id\` BIGINT UNSIGNED NOT NULL COMMENT '所属任务ID（外键→task.id）',

&nbsp;   \`source_type\` ENUM('GLOBAL', 'UPLOAD') NOT NULL COMMENT '数据来源：GLOBAL（引用全局文件库）/ UPLOAD（本地上传）',

&nbsp;   \`global_doc_id\` BIGINT UNSIGNED DEFAULT NULL COMMENT '引用全局文件ID（外键→global_document.id）',

&nbsp;   \`file_name\` VARCHAR(255) DEFAULT NULL COMMENT '原始文件名（UPLOAD时使用）',

&nbsp;   \`file_path\` VARCHAR(500) DEFAULT NULL COMMENT '上传文件存储路径（UPLOAD时使用）',

&nbsp;   \`extracted_text\` LONGTEXT NOT NULL COMMENT '最终用于标注的纯文本',

&nbsp;   \`extracted_range\` VARCHAR(100) DEFAULT NULL COMMENT '自动提取范围描述',

&nbsp;   \`is_manually_corrected\` TINYINT NOT NULL DEFAULT 0 COMMENT '是否手动修正过提取范围：1是 0否',

&nbsp;   \`uploaded_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '添加时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_task\` (\`task_id\`),

&nbsp;   KEY \`idx_global\` (\`global_doc_id\`),

&nbsp;   CONSTRAINT \`fk_td_task\` FOREIGN KEY (\`task_id\`) REFERENCES \`task\` (\`id\`) ON DELETE CASCADE,

&nbsp;   CONSTRAINT \`fk_td_global\` FOREIGN KEY (\`global_doc_id\`) REFERENCES \`global_document\` (\`id\`) ON DELETE SET NULL

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='任务文档表';

\-- 任务分配表

CREATE TABLE \`task_assignment\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '分配记录ID（主键）',

&nbsp;   \`task_id\` BIGINT UNSIGNED NOT NULL COMMENT '任务ID（外键→task.id）',

&nbsp;   \`document_id\` BIGINT UNSIGNED NOT NULL COMMENT '文档ID（外键→task_document.id）',

&nbsp;   \`user_id\` BIGINT UNSIGNED NOT NULL COMMENT '用户ID（外键→sys_user.id）',

&nbsp;   \`role_in_task\` VARCHAR(20) NOT NULL COMMENT '任务内身份：标注员/裁定者',

&nbsp;   \`assigned_by\` BIGINT UNSIGNED NOT NULL COMMENT '分配者ID（外键→sys_user.id）',

&nbsp;   \`assigned_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '分配时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   UNIQUE KEY \`uk_task_doc_user_role\` (\`task_id\`, \`document_id\`, \`user_id\`, \`role_in_task\`),

&nbsp;   KEY \`idx_user\` (\`user_id\`),

&nbsp;   KEY \`idx_task_role\` (\`task_id\`, \`role_in_task\`),

&nbsp;   CONSTRAINT \`fk_ta_task\` FOREIGN KEY (\`task_id\`) REFERENCES \`task\` (\`id\`) ON DELETE CASCADE,

&nbsp;   CONSTRAINT \`fk_ta_document\` FOREIGN KEY (\`document_id\`) REFERENCES \`task_document\` (\`id\`) ON DELETE CASCADE,

&nbsp;   CONSTRAINT \`fk_ta_user\` FOREIGN KEY (\`user_id\`) REFERENCES \`sys_user\` (\`id\`) ON DELETE RESTRICT,

&nbsp;   CONSTRAINT \`fk_ta_assigned_by\` FOREIGN KEY (\`assigned_by\`) REFERENCES \`sys_user\` (\`id\`) ON DELETE RESTRICT

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='任务分配表';

\-- ==================== 标注核心模块 ====================

\-- 命题表

CREATE TABLE \`proposition\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '命题ID（主键）',

&nbsp;   \`document_id\` BIGINT UNSIGNED NOT NULL COMMENT '所属文档ID（外键→task_document.id）',

&nbsp;   \`sequence_no\` INT NOT NULL COMMENT '在文档中的序号',

&nbsp;   \`start_pos\` INT NOT NULL COMMENT '起始字符位置',

&nbsp;   \`end_pos\` INT NOT NULL COMMENT '结束字符位置',

&nbsp;   \`selected_text\` TEXT NOT NULL COMMENT '选中的文本片段',

&nbsp;   \`label_path\` VARCHAR(50) NOT NULL COMMENT '标签路径',

&nbsp;   \`created_by\` BIGINT UNSIGNED NOT NULL COMMENT '创建者ID（外键→sys_user.id）',

&nbsp;   \`created_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

&nbsp;   \`updated_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_document\` (\`document_id\`),

&nbsp;   KEY \`idx_sequence\` (\`document_id\`, \`sequence_no\`),

&nbsp;   CONSTRAINT \`fk_prop_doc\` FOREIGN KEY (\`document_id\`) REFERENCES \`task_document\` (\`id\`) ON DELETE CASCADE,

&nbsp;   CONSTRAINT \`fk_prop_user\` FOREIGN KEY (\`created_by\`) REFERENCES \`sys_user\` (\`id\`) ON DELETE RESTRICT

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='命题表';

\-- 关系表

CREATE TABLE \`relation\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '关系ID（主键）',

&nbsp;   \`document_id\` BIGINT UNSIGNED NOT NULL COMMENT '所属文档ID（外键→task_document.id）',

&nbsp;   \`relation_type\` CHAR(1) NOT NULL COMMENT '关系类型',

&nbsp;   \`is_nested\` TINYINT NOT NULL DEFAULT 0 COMMENT '是否为嵌套关系：1是 0否',

&nbsp;   \`parent_relation_id\` BIGINT UNSIGNED DEFAULT NULL COMMENT '父关系ID（嵌套时外键→relation.id）',

&nbsp;   \`created_by\` BIGINT UNSIGNED NOT NULL COMMENT '创建者ID（外键→sys_user.id）',

&nbsp;   \`created_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',

&nbsp;   \`updated_at\` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_document\` (\`document_id\`),

&nbsp;   KEY \`idx_parent\` (\`parent_relation_id\`),

&nbsp;   CONSTRAINT \`fk_rel_doc\` FOREIGN KEY (\`document_id\`) REFERENCES \`task_document\` (\`id\`) ON DELETE CASCADE,

&nbsp;   CONSTRAINT \`fk_rel_parent\` FOREIGN KEY (\`parent_relation_id\`) REFERENCES \`relation\` (\`id\`) ON DELETE CASCADE,

&nbsp;   CONSTRAINT \`fk_rel_user\` FOREIGN KEY (\`created_by\`) REFERENCES \`sys_user\` (\`id\`) ON DELETE RESTRICT

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='关系表';

\-- 关系成员明细表

CREATE TABLE \`relation_member\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '成员明细ID（主键）',

&nbsp;   \`relation_id\` BIGINT UNSIGNED NOT NULL COMMENT '所属关系ID（外键→relation.id）',

&nbsp;   \`member_type\` ENUM('P', 'R') NOT NULL COMMENT '成员类型：P=命题，R=关系',

&nbsp;   \`member_id\` BIGINT UNSIGNED NOT NULL COMMENT '成员ID（命题.id 或 关系.id）',

&nbsp;   \`member_role\` VARCHAR(10) NOT NULL DEFAULT 'MEMBER' COMMENT '角色：SOURCE/TARGET/MEMBER',

&nbsp;   \`member_order\` INT NOT NULL DEFAULT 0 COMMENT '参数顺序',

&nbsp;   PRIMARY KEY (\`id\`),

&nbsp;   KEY \`idx_relation\` (\`relation_id\`),

&nbsp;   KEY \`idx_member\` (\`member_type\`, \`member_id\`),

&nbsp;   CONSTRAINT \`fk_rm_relation\` FOREIGN KEY (\`relation_id\`) REFERENCES \`relation\` (\`id\`) ON DELETE CASCADE

) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='关系成员明细表';

\-- 标注结果快照表

CREATE TABLE \`annotation\` (

&nbsp;   \`id\` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT COMMENT '标注结果ID（主键）',

&nbsp;   \`task_id\` BIGINT UNSIGNED NOT NULL COMMENT '所属任务ID（外键→task.id）',

&nbsp;   \`document_id\` BIGINT UNSIGNED NOT NULL COMMENT '文档ID（外键→task_document.id）',

&nbsp;   \`user_id\` BIGINT UNSIGNED NOT NULL COMMENT '标注员ID（外键→sys_user.id）',

&nbsp;   \`proposition_data\` JSON NOT NULL COMMENT '该版本所有命题数据的快照',

&nbsp;   \`relation_data\` JSON NOT NULL COMMENT '该版本所有关系数据的快照',
